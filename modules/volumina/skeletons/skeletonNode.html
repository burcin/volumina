[docs]    def intersectsBbox(self, point):
        assert len(point) == 3
        for i in range(3):
            if not (self.pos[i] - self.shape/2.0 >= point[i] and self.pos[i] + self.shape/2.0 <= point[i]):
                return False 
        return True

[docs]    def pointF(self, axis=None):
        if axis is None:
            axis = self.axis
        pos2D = list(self.pos)
        del pos2D[axis]
        return QPointF(*pos2D)
